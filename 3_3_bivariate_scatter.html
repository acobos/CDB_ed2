library(tidyverse)
library(ggformula)

d <- rio::import("data/OSAS.xls") %>% 
    mutate(gender = factor(gender, 1:0, c("male", "female")),
           ftp = factor(ftp, 1:4, paste("grade", 1:4)),
           tonsils = factor(tonsils, 0:4, paste("grade", 0:4)),
           uvula = factor (uvula, 1:4, c("no hypertrophy", "slight hyp.",
